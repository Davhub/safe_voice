// Firebase Firestore Security Rules for Anonymous Reporting App
// These rules should be applied in the Firebase Console -> Firestore -> Rules

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    // Allow anonymous users to create reports
    match /reports/{document} {
      allow create: if true; // Allow anonymous report creation
      allow read, update, delete: if false; // Prevent reading/modifying for privacy
    }
    
    // Deny all other access
    match /{document=**} {
      allow read, write: if false;
    }
  }
}
